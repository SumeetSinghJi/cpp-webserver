cmake_minimum_required(VERSION 3.0)

project(cpp-webserver)

set(CMAKE_LIBRARIES_DIR
    "-lws2_32"
)

set(SOURCES 
    main.cpp
    lightweight_cpp_webserver.cpp
)

include_directories(
    headers
    website
)

add_executable(cpp-webserver ${SOURCES})

target_include_directories(cpp-webserver PRIVATE "${CMAKE_SOURCE_DIR}/headers")
target_link_libraries(cpp-webserver PRIVATE "${CMAKE_LIBRARIES_DIR}")
