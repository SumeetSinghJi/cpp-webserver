cmake_minimum_required(VERSION 3.0)

project(cpp-webserver)

find_package(OpenSSL REQUIRED)

set(SOURCES 
    main.cpp
    lightweight_cpp_webserver.cpp
)

include_directories(
    headers
    website
    ${OPENSSL_INCLUDE_DIR}  # Include OpenSSL header directory
)

add_executable(cpp-webserver ${SOURCES})

target_link_libraries(cpp-webserver PRIVATE ${OPENSSL_LIBRARIES})
